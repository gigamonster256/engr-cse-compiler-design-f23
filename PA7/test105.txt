// nested if/while testcase and TAC IR

main

int a, b, c, d, e;

{
    a = 1;
    b = 2;
    c = 3;
    d = 4;
    e = 5;

    while ((c+4) < d) do
        if ((a+5-b) > (b*6)) then
            a = a / c * d;
            d = d - 1;
            call printInt(a);
            call printInt(d);
        else
            d = a + b;
            call printInt(d);
        fi;
        call println();
    od;

    e = c - a + b;
    call printInt(e); // 4
    a = a + b;
}.
